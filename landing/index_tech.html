  playerName: string;
  score: number;
  socialBattery: number;      // 0-100
  anxietyLevel: number;        // 0-100
  phase: Phase;                // FSM
  currentLevel: number;        // 1-9
  gameOverReason: GameOverReason;
  gameWon: boolean;

Game Loop (useFrame)

useFrame((state, delta) => {
  // 1. INPUT
  const keys = getKeys();
  const focusActive = keys.focus && phase === 'approach';
  
  // 2. TIME DILATION
  const timeScale = focusActive ? 0.2 : 1.0;
  
  // 3. SIMULATION
  if (phase === 'approach') {
    tickSimulation(delta); // Bater√≠a + Ansiedad
  }
  
  // 4. MOVEMENT
  if (keys.forward) targetPos.z -= speed * timeScale;
  
  // 5. COLLISION
  if (hitboxOverlap(player, car)) {
    hitObstacle(); // GAME OVER
  }
  
  // 6. CAMERA
  camera.position.lerp(targetPos, 0.1);
});

Dynamic Post-Processing

const DynamicPostProcessing = () => {
  const anxietyLevel = useGameStore(state => state.anxietyLevel);
  
  // Map 0-100 anxiety to effects
  const aberration = (anxietyLevel / 100) * 0.01;
  const vignette = 0.8 + (anxietyLevel / 100) * 0.5;
  const noise = 0.02 + (anxietyLevel / 100) * 0.1;
  
  return (
    <EffectComposer>
      <Bloom />
      <ChromaticAberration offset={aberration} />
      <Vignette darkness={vignette} />
      <Noise opacity={noise} />
    </EffectComposer>
  );
};
